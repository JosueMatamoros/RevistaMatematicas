%  \trxn{x}{n}: expresión directa; #2 es un literal cuando se expande 
%    T_n(x) = (x/n)*[ (1 + exp(-x^2))/2 + sum_{k=1}^{n-1} exp(-(k*x/n)^2) ]
\newcommand{\trxn}[2]{%
	\num{%
		\xintthefloatexpr
		(#1 / #2) * ( (1 + exp(-(#1)^2)) / 2
		+ add( exp(-((k * (#1/#2))^2)), k = 1 .. (#2-1) ) )
		\relax
	}%
}
